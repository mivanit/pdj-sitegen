pdj_sitegen.build

Build a site from markdown files using Jinja2 and Pandoc

Pipeline:

  • Get all markdown files
  • For each markdown file:
    • Regex frontmatter and content from the markdown file
    • Execute a template on the frontmatter, with globals_ and file metadata as context
    • Load the frontmatter into a dict
    • Execute a template on the content with frontmatter, globals_, file metadata, and all other docs as context
    • Convert the content to HTML using Pandoc
    • Execute a template on the specified or default template with the HTML content, frontmatter, globals_ and file metadata as context
  • Copy the resources directory to the output directory

 47def split_md(
 48	content: str,
 49) -> Tuple[str, str, Format]:
 50	"""parse markdown into a tuple of frontmatter, body, and frontmatter format
 51
 52	will use `FRONTMATTER_REGEX` to split the markdown content into frontmatter and body.
 53	the possible delimiters are defined in `FRONTMATTER_DELIMS`.
 54
 55	# Parameters:
 56	 - `content : str`
 57	   markdown content to split
 58
 59	# Returns:
 60	 - `Tuple[str, str, Format]`
 61	   tuple of frontmatter, body, and frontmatter format (yaml, json, toml)
 62
 63	# Raises:
 64	 - `SplitMarkdownError` : if the regex does not match
 65	"""
 66	match: Optional[re.Match[str]] = re.match(FRONTMATTER_REGEX, content)
 67	frontmatter: str
 68	body: str
 69	fmt: Format
 70	if match:
 71		# if the regex matches, extract the frontmatter, body, and format
 72		delimiter: str = match.group("delimiter")
 73		frontmatter = match.group("frontmatter")
 74		assert isinstance(frontmatter, str)
 75		body = match.group("body")
 76		assert isinstance(body, str)
 77		fmt = FRONTMATTER_DELIMS[delimiter]
 78	else:
 79		raise SplitMarkdownError(f"No frontmatter found in content\n{content = }")
 80
 81	return frontmatter, body, fmt

 84def render(
 85	content: str,
 86	context: dict[str, Any],
 87	jinja_env: Environment,
 88) -> str:
 89	"""render content given context and jinja2 environment. raise RenderError if error occurs
 90
 91	# Parameters:
 92	 - `content : str`
 93	   text content with jinja2 template syntax
 94	 - `context : dict[str, Any]`
 95	   data to render into the template
 96	 - `jinja_env : Environment`
 97	   jinja2 environment to use for rendering
 98
 99	# Returns:
100	 - `str`
101	   rendered content
102
103	# Raises:
104	 - `RenderError` : if an error occurs while creating or rendering the template
105	"""
106	try:
107		template: Template = jinja_env.from_string(content)
108	except Exception as e_template:
109		raise RenderError(
110			"Error creating template",
111			kind="create_template",
112			content=content,
113			context=context,
114			jinja_env=jinja_env,
115			template=None,
116		) from e_template
117
118	try:
119		output: str = template.render(context)
120	except Exception as e_render:
121		raise RenderError(
122			"Error rendering template",
123			kind="render_template",
124			content=content,
125			context=context,
126			jinja_env=jinja_env,
127			template=template,
128		) from e_render
129
130	return output

133def build_document_tree(
134	content_dir: Path,
135	frontmatter_context: dict[str, Any],
136	jinja_env: Environment,
137	verbose: bool = True,
138) -> dict[str, dict[str, Any]]:
139	"""given a dir of markdown files, return a dict of documents with rendered frontmatter
140
141	documents are keyed by their path relative to `content_dir`, with suffix removed.
142	the dict for each document will contain:
143
144	- `frontmatter: dict[str, Any]` : rendered and parsed frontmatter for that document
145	- `body: str` : plain, unrendered markdown content for that document
146	- `file_meta: dict[str, Any]` : metadata about the file, including `"path", "path_html", "path_raw", "modified_time"`
147
148	# Parameters:
149	 - `content_dir : Path`
150	   path to glob for markdown files
151	 - `frontmatter_context : dict[str, Any]`
152	   context to use to render the frontmatter *before* parsing it into a dict
153	 - `jinja_env : Environment`
154	   jinja2 environment to use for rendering
155
156	# Returns:
157	 - `dict[str, dict[str, Any]]`
158	   dict of documents with rendered frontmatter.
159	"""
160	md_files: list[Path] = list(content_dir.rglob("*.md"))
161
162	if verbose:
163		print(f"Found {len(md_files)} markdown files in '{content_dir}'")
164
165	docs: dict[str, dict[str, Any]] = {}
166
167	for file_path in tqdm.tqdm(
168		md_files,
169		desc="building document tree",
170		unit="file",
171		disable=not verbose,
172	):
173		file_path_str: str = (
174			file_path.relative_to(content_dir).as_posix().removesuffix(".md")
175		)
176		with open(file_path, "r", encoding="utf-8") as f:
177			content: str = f.read()
178		frontmatter_raw: str
179		body: str
180		fmt: Format
181		frontmatter_raw, body, fmt = split_md(content)
182
183		last_modified_time: float = file_path.stat().st_mtime
184		file_meta: dict[str, Any] = {
185			"path": file_path_str,
186			"path_html": f"{file_path_str}.html",
187			"path_raw": file_path.as_posix(),
188			"modified_time": last_modified_time,
189			"modified_time_str": datetime.datetime.fromtimestamp(
190				last_modified_time
191			).strftime("%Y-%m-%d %H:%M:%S"),
192		}
193
194		frontmatter_rendered: str = render(
195			content=frontmatter_raw,
196			context={**frontmatter_context, "file_meta": file_meta},
197			jinja_env=jinja_env,
198		)
199		frontmatter: dict[str, Any] = FORMAT_PARSERS[fmt](frontmatter_rendered)
200
201		docs[file_path_str] = {
202			"frontmatter": frontmatter,
203			"body": body,
204			"file_meta": file_meta,
205		}
206	return docs

209def process_pandoc_args(pandoc_args: dict[str, Any]) -> list[str]:
210	"""given args to pass to pandoc, turn them into a list of strings we can actually pass
211
212	keys must be strings. values can be strings, bools, or iterables of strings.
213
214	when a value is a:
215
216	- `bool` : if True, add the key to the list. if False, skip it.
217	- `str` : add the key and value to the list together.
218	- `iterable` : for each item in the iterable, add the key and item to the list together.
219	                (i.e. `"filters": ["filter_a", "filter_b"]` -> `["--filters", "filter_a", "--filters", "filter_b"]`)
220
221	# Parameters:
222	 - `pandoc_args : dict[str, Any]`
223
224	# Returns:
225	 - `list[str]`
226	"""
227	args: list[str] = list()
228	for k, v in pandoc_args.items():
229		if isinstance(v, bool):
230			if v:
231				args.append(f"--{k}")
232		elif isinstance(v, str):
233			args.extend([f"--{k}", v])
234		elif isinstance(v, Iterable):
235			for x in v:
236				args.extend([f"--{k}", x])
237		else:
238			raise ValueError(f"Invalid type for pandoc arg: {type(v) = } {v = }")
239
240	return args

246	fmt: str,
247	path: str,

249) -> None:

256		with open(output_path, "w", encoding="utf-8") as f:
257			f.write(content)

260def convert_single_markdown_file(
261	path: str,
262	output_root: Path,
263	doc: dict[str, Any],
264	docs: dict[str, dict[str, Any]],
265	jinja_env: Environment,
266	config: Config,
267	intermediates_dir: Optional[Path] = None,
268) -> None:
269	frontmatter: dict = doc.get("frontmatter", {})
270	assert isinstance(frontmatter, dict)
271	body: str = doc.get("body", "")
272	assert isinstance(body, str)
273	file_meta: dict = doc.get("file_meta", {})
274	assert isinstance(file_meta, dict)
275	context: dict[str, Any] = {
276		**frontmatter,
277		"frontmatter": frontmatter,
278		"file_meta": file_meta,
279		"config": config.serialize(),
280		"docs": docs,
281		"child_docs": {
282			k: v for k, v in docs.items() if (k.startswith(path) and k != path)
283		},
284	}
285
286	dump_intermediate_partial: Callable = functools.partial(
287		dump_intermediate,
288		intermediates_dir=intermediates_dir,
289		path=file_meta["path"],
290	)
291
292	# dump frontmatter to intermediates
293	dump_intermediate_partial(
294		content=str(frontmatter),
295		fmt="txt",
296		subdir="frontmatter_txt",
297	)
298	dump_intermediate_partial(
299		content=json.dumps(json_serialize(frontmatter)),
300		fmt="json",
301		subdir="frontmatter_json",
302	)
303
304	# Now, execute a template on the content with context
305	# Render Markdown content with Jinja2
306	rendered_md: str = render(
307		content=body,
308		context=context,
309		jinja_env=jinja_env,
310	)
311
312	dump_intermediate_partial(content=rendered_md, fmt="md")
313
314	# Convert Markdown to HTML using Pandoc
315	pandoc_args: list[str] = process_pandoc_args(
316		{
317			**config.pandoc_kwargs,
318			**context["frontmatter"].get("pandoc_kwargs", {}),
319		}
320	)
321
322	html_content: str = pypandoc.convert_text(
323		source=rendered_md,
324		to=config.pandoc_fmt_to,
325		format=config.pandoc_fmt_from,
326		extra_args=pandoc_args,
327	)
328
329	dump_intermediate_partial(content=html_content, fmt="html")
330
331	# Determine which HTML template to use
332	template_name: str = frontmatter.get(
333		"__template__",
334		config.default_template.as_posix(),
335	)
336
337	# Render final HTML
338	template: Template = jinja_env.get_template(template_name)
339	final_html: str = template.render({"__content__": html_content, **context})
340
341	# Output HTML file
342	output_path: Path = output_root / config.output_dir / file_meta["path_html"]
343	output_path.parent.mkdir(parents=True, exist_ok=True)
344	with open(output_path, "w", encoding="utf-8") as f:
345		f.write(final_html)

348def convert_markdown_files(
349	docs: dict[str, dict[str, Any]],
350	jinja_env: Environment,
351	config: Config,
352	output_root: Path,
353	smart_rebuild: bool,
354	rebuild_time: float,
355	verbose: bool = True,
356	intermediates_dir: Optional[Path] = None,
357) -> None:
358	n_files: int = len(docs)
359	path: str
360	doc: dict[str, Any]
361	exceptions: dict[str, Exception] = dict()
362	if verbose:
363		print(f"Converting {n_files} markdown files to HTML...")
364	for idx, (path, doc) in enumerate(docs.items()):
365		path_raw: str = doc["file_meta"]["path_raw"]
366		if smart_rebuild and os.path.getmtime(path_raw) <= rebuild_time:
367			if verbose:
368				print(f"\t({idx+1:3} / {n_files})  [unmodified]  '{path_raw}'")
369		else:
370			if verbose:
371				print(f"\t({idx+1:3} / {n_files})  [building..]  '{path_raw}'")
372
373			try:
374				convert_single_markdown_file(
375					path=path,
376					output_root=output_root,
377					doc=doc,
378					docs=docs,
379					jinja_env=jinja_env,
380					config=config,
381					intermediates_dir=intermediates_dir,
382				)
383			except Exception as e:
384				exceptions[path_raw] = e
385				if verbose:
386					print(f"\t  Error converting '{path_raw}'!!!")
387	if exceptions:
388		first_key: str = next(iter(exceptions.keys()))
389		if len(exceptions) == 1:
390			raise ConversionError(
391				f"error converting file '{first_key}'\n{exceptions[first_key]}"
392			) from exceptions[first_key]
393		else:
394			raise MultipleExceptions(
395				f"failed to convert {len(exceptions)}/{n_files} files",
396				exceptions,
397			) from exceptions[first_key]

400def pipeline(
401	config_path: Path,
402	verbose: bool = True,
403	smart_rebuild: bool = False,
404) -> None:
405	"""build the website
406
407	# what this does:
408
409	- change directory to the directory containing the config file
410	- read the config file from the given path
411	- set up a Jinja2 environment according to the config
412	- build a document tree from the markdown files in the content directory
413	- process the markdown files into HTML files and write them to the output directory
414	"""
415
416	# set up spinner context manager, depending on verbosity
417	sp_class: type[ContextManager] = (
418		functools.partial(SpinnerContext, update_interval=0.01)  # type: ignore[assignment]
419		if verbose
420		else NoOpContextManager
421	)
422
423	# get config path, change to the directory containing the config file
424	root_dir: Path = config_path.parent
425	root_dir_absolute: Path = root_dir.absolute()
426
427	# read config and set up Jinja environment
428	with sp_class(message="read config and set up jinja environment..."):  # type: ignore[call-arg]
429		# Read the config file
430		config: Config = Config.read(root_dir_absolute / config_path.name)
431
432		# Set up Jinja2 environment
433		jinja_env = Environment(
434			loader=FileSystemLoader([root_dir_absolute / config.templates_dir]),
435			**config.jinja_env_kwargs,
436		)
437
438		# figure out the last rebuild time, write the current time to the file
439		rebuild_time: float
440		try:
441			rebuild_time = os.path.getmtime(config.build_time_fname)
442		except FileNotFoundError:
443			# set it to very old time so that all files are rebuilt
444			rebuild_time = -1.0
445
446		with open(config.build_time_fname, "w", encoding="utf-8") as f:
447			f.write(str(rebuild_time))
448
449	# build doc tree (get .md files from `config.content_dir`, split content and frontmatter, execute templates on frontmatter)
450	docs: dict[str, dict[str, Any]] = build_document_tree(
451		content_dir=root_dir_absolute / config.content_dir,
452		frontmatter_context={"config": config.serialize()},
453		jinja_env=jinja_env,
454		verbose=verbose,
455	)
456
457	# convert markdown files to HTML (execute templates with frontmatter on content, convert to HTML with Pandoc, execute template on HTML)
458	convert_markdown_files(
459		docs=docs,
460		jinja_env=jinja_env,
461		config=config,
462		output_root=root_dir_absolute,
463		smart_rebuild=smart_rebuild,
464		rebuild_time=rebuild_time,
465		verbose=verbose,
466		intermediates_dir=(
467			root_dir_absolute / config.intermediates_dir
468			if config.intermediates_dir
469			else None
470		),
471	)
472
473	# copy resources dir to output dir
474	with sp_class(message="Copying resources directory..."):  # type: ignore[call-arg]
475		src_abs: Path = root_dir_absolute / config.content_dir / config.resources_dir
476		out_abs: Path = root_dir_absolute / config.output_dir / config.resources_dir
477		shutil.copytree(
478			src=src_abs,
479			dst=out_abs,
480			dirs_exist_ok=True,
481		)
482

484def main() -> None:
485	# parse args
486	arg_parser: argparse.ArgumentParser = argparse.ArgumentParser()
487	arg_parser.add_argument("config_path", type=str, help="path to the config file")
488	arg_parser.add_argument(
489		"-q",
490		"--quiet",
491		action="store_true",
492		help="disable verbose output",
493	)
494	arg_parser.add_argument(
495		"-s",
496		"--smart-rebuild",
497		action="store_true",
498		help="enable smart rebuild",
499	)
500	args: argparse.Namespace = arg_parser.parse_args()
501	pipeline(
502		config_path=Path(args.config_path),
503		verbose=not args.quiet,
504		smart_rebuild=args.smart_rebuild,
505	)
506
507
508if __name__ == "__main__":
509	main()

235		elif isinstance(v, Iterable):
236			for x in v:
