pdj_sitegen.build

Build a site from markdown files using Jinja2 and Pandoc

Pipeline:

  • Get all markdown files
  • For each markdown file:
    • Regex frontmatter and content from the markdown file
    • Execute a template on the frontmatter, with globals_ and file metadata as context
    • Load the frontmatter into a dict
    • Execute a template on the content with frontmatter, globals_, file metadata, and all other docs as context
    • Convert the content to HTML using Pandoc
    • Execute a template on the specified or default template with the HTML content, frontmatter, globals_ and file metadata as context
  • Copy content files to output directory (based on copy_include/copy_exclude patterns)

 51def should_copy(rel_path: str, include: list[str], exclude: list[str]) -> bool:
 52	"""Determine if a file should be copied based on include/exclude patterns.
 53
 54	Rules:
 55	- If file matches any include pattern → copy (wins over exclude)
 56	- If file matches any exclude pattern → don't copy
 57	- If include is empty → copy everything not excluded
 58	- If include is non-empty but file doesn't match → don't copy
 59
 60	# Parameters:
 61	 - `rel_path : str`
 62	   relative path of the file (POSIX format)
 63	 - `include : list[str]`
 64	   glob patterns for files to include (empty means everything)
 65	 - `exclude : list[str]`
 66	   glob patterns for files to exclude
 67
 68	# Returns:
 69	 - `bool`
 70	   True if the file should be copied
 71	"""
 72	matches_include = any(fnmatch.fnmatch(rel_path, p) for p in include)
 73	matches_exclude = any(fnmatch.fnmatch(rel_path, p) for p in exclude)
 74
 75	if matches_include:
 76		return True  # Explicit include wins over exclude
 77	if matches_exclude:
 78		return False
 79	return not include  # Empty include = copy everything not excluded

 82def copy_content_files(
 83	content_dir: Path,
 84	output_dir: Path,
 85	include: list[str],
 86	exclude: list[str],
 87	verbose: bool = True,
 88) -> int:
 89	"""Copy files from content_dir to output_dir based on include/exclude patterns.
 90
 91	# Parameters:
 92	 - `content_dir : Path`
 93	   source directory containing content files
 94	 - `output_dir : Path`
 95	   destination directory for output
 96	 - `include : list[str]`
 97	   glob patterns for files to include (empty means everything)
 98	 - `exclude : list[str]`
 99	   glob patterns for files to exclude
100	 - `verbose : bool`
101	   whether to print progress information
102
103	# Returns:
104	 - `int`
105	   number of files copied
106	"""
107	copied_count = 0
108	for file_path in content_dir.rglob("*"):
109		if file_path.is_file():
110			rel_path = file_path.relative_to(content_dir).as_posix()
111			if should_copy(rel_path, include, exclude):
112				dest = output_dir / rel_path
113				dest.parent.mkdir(parents=True, exist_ok=True)
114				shutil.copy2(file_path, dest)
115				copied_count += 1
116	if verbose:
117		print(f"Copied {copied_count} resource files")
118	return copied_count

121def split_md(
122	content: str,
123) -> tuple[str, str, Format]:
124	"""parse markdown into a tuple of frontmatter, body, and frontmatter format
125
126	will use `FRONTMATTER_REGEX` to split the markdown content into frontmatter and body.
127	the possible delimiters are defined in `FRONTMATTER_DELIMS`.
128
129	# Parameters:
130	 - `content : str`
131	   markdown content to split
132
133	# Returns:
134	 - `tuple[str, str, Format]`
135	   tuple of frontmatter, body, and frontmatter format (yaml, json, toml)
136
137	# Raises:
138	 - `SplitMarkdownError` : if the regex does not match
139	"""
140	match: re.Match[str] | None = re.match(FRONTMATTER_REGEX, content)
141	frontmatter: str
142	body: str
143	fmt: Format
144	if match:
145		# if the regex matches, extract the frontmatter, body, and format
146		delimiter: str = match.group("delimiter")
147		frontmatter = match.group("frontmatter")
148		if not isinstance(frontmatter, str):
149			raise TypeError(f"Expected frontmatter to be str, got {type(frontmatter)}")
150		body = match.group("body")
151		if not isinstance(body, str):
152			raise TypeError(f"Expected body to be str, got {type(body)}")
153		fmt = FRONTMATTER_DELIMS[delimiter]
154	else:
155		raise SplitMarkdownError(f"No frontmatter found in content\n{content = }")
156
157	return frontmatter, body, fmt

160def render(
161	content: str,
162	context: dict[str, Any],
163	jinja_env: Environment,
164) -> str:
165	"""render content given context and jinja2 environment. raise RenderError if error occurs
166
167	# Parameters:
168	 - `content : str`
169	   text content with jinja2 template syntax
170	 - `context : dict[str, Any]`
171	   data to render into the template
172	 - `jinja_env : Environment`
173	   jinja2 environment to use for rendering
174
175	# Returns:
176	 - `str`
177	   rendered content
178
179	# Raises:
180	 - `RenderError` : if an error occurs while creating or rendering the template
181	"""
182	try:
183		template: Template = jinja_env.from_string(content)
184	except Exception as e_template:
185		raise RenderError(
186			"Error creating template",
187			kind="create_template",
188			content=content,
189			context=context,
190			jinja_env=jinja_env,
191			template=None,
192		) from e_template
193
194	try:
195		output: str = template.render(context)
196	except Exception as e_render:
197		raise RenderError(
198			"Error rendering template",
199			kind="render_template",
200			content=content,
201			context=context,
202			jinja_env=jinja_env,
203			template=template,
204		) from e_render
205
206	return output

209def build_document_tree(
210	content_dir: Path,
211	frontmatter_context: dict[str, Any],
212	jinja_env: Environment,
213	verbose: bool = True,
214	normalize_index_names: bool = True,
215) -> dict[str, dict[str, Any]]:
216	"""given a dir of markdown files, return a dict of documents with rendered frontmatter
217
218	documents are keyed by their path relative to `content_dir`, with suffix removed.
219	the dict for each document will contain:
220
221	- `frontmatter: dict[str, Any]` : rendered and parsed frontmatter for that document
222	- `body: str` : plain, unrendered markdown content for that document
223	- `file_meta: dict[str, Any]` : metadata about the file, including `"path", "path_html", "path_raw", "modified_time"`
224
225	# Parameters:
226	 - `content_dir : Path`
227	   path to glob for markdown files
228	 - `frontmatter_context : dict[str, Any]`
229	   context to use to render the frontmatter *before* parsing it into a dict
230	 - `jinja_env : Environment`
231	   jinja2 environment to use for rendering
232	 - `normalize_index_names : bool`
233	   if True, `_index.md` files are renamed to `index.html` in output
234
235	# Returns:
236	 - `dict[str, dict[str, Any]]`
237	   dict of documents with rendered frontmatter.
238
239	# Raises:
240	 - `ConflictingIndexError` : if both `index.md` and `_index.md` exist in the same directory
241	"""
242	md_files: list[Path] = list(content_dir.rglob("*.md"))
243
244	# Check for conflicting index files when normalization is enabled
245	if normalize_index_names:
246		# Group files by parent directory
247		dirs_with_index: dict[Path, list[str]] = {}
248		for file_path in md_files:
249			if file_path.stem in ("index", "_index"):
250				parent = file_path.parent
251				if parent not in dirs_with_index:
252					dirs_with_index[parent] = []
253				dirs_with_index[parent].append(file_path.name)
254
255		# Check for conflicts
256		for directory, index_files in dirs_with_index.items():
257			if len(index_files) > 1:
258				rel_dir = directory.relative_to(content_dir).as_posix()
259				if rel_dir == ".":
260					rel_dir = "(root)"
261				raise ConflictingIndexError(rel_dir, sorted(index_files))
262
263	if verbose:
264		print(f"Found {len(md_files)} markdown files in '{content_dir}'")
265
266	docs: dict[str, dict[str, Any]] = {}
267	errors: dict[str, Exception] = {}
268
269	for file_path in tqdm.tqdm(
270		md_files,
271		desc="building document tree",
272		unit="file",
273		disable=not verbose,
274	):
275		file_path_str: str = (
276			file_path.relative_to(content_dir).as_posix().removesuffix(".md")
277		)
278
279		# Compute output path, normalizing _index -> index if enabled
280		output_path_str: str = file_path_str
281		if normalize_index_names and file_path.stem == "_index":
282			# Replace _index with index in the output path
283			parent_str = file_path.parent.relative_to(content_dir).as_posix()
284			if parent_str == ".":
285				output_path_str = "index"
286			else:
287				output_path_str = f"{parent_str}/index"
288
289		with open(file_path, "r", encoding="utf-8") as f:
290			content: str = f.read()
291
292		try:
293			frontmatter_raw: str
294			body: str
295			fmt: Format
296			frontmatter_raw, body, fmt = split_md(content)
297
298			last_modified_time: float = file_path.stat().st_mtime
299			file_meta: dict[str, Any] = {
300				"path": file_path_str,
301				"path_stem": file_path.stem,
302				"path_html": f"{output_path_str}.html",
303				"path_raw": file_path.as_posix(),
304				"path_to_root": "/".join([".."] * file_path_str.count("/")) or ".",
305				"modified_time": last_modified_time,
306				"modified_time_str": datetime.datetime.fromtimestamp(
307					last_modified_time
308				).strftime("%Y-%m-%d %H:%M:%S"),
309			}
310
311			frontmatter_rendered: str = render(
312				content=frontmatter_raw,
313				context={**frontmatter_context, "file_meta": file_meta},
314				jinja_env=jinja_env,
315			)
316			frontmatter: dict[str, Any] = FORMAT_PARSERS[fmt](frontmatter_rendered)
317
318			docs[file_path_str] = {
319				"frontmatter": frontmatter,
320				"body": body,
321				"file_meta": file_meta,
322			}
323		except SplitMarkdownError as e:
324			# Add file path context if not already set
325			if not e.file_path:
326				e = SplitMarkdownError(str(e), file_path=file_path.as_posix())
327			errors[file_path.as_posix()] = e
328			continue
329
330	if errors:
331		print(
332			f"\nMissing frontmatter in {len(errors)} file(s):",
333			file=sys.stderr,
334		)
335		for filepath in errors:
336			print(f"  - {filepath}", file=sys.stderr)
337		raise MultipleExceptions(
338			f"Missing frontmatter in {len(errors)}/{len(md_files)} files",
339			exceptions=errors,
340			n_total=len(md_files),
341		)
342
343	return docs

366def process_pandoc_args(pandoc_args: dict[str, Any]) -> list[str]:
367	"""given args to pass to pandoc, turn them into a list of strings we can actually pass
368
369	keys must be strings. values can be strings, bools, or iterables of strings.
370
371	when a value is a:
372
373	- `bool` : if True, add the key to the list. if False, skip it.
374	- `str` : add the key and value to the list together.
375	- `iterable` : for each item in the iterable, add the key and item to the list together.
376	                (i.e. `"filters": ["filter_a", "filter_b"]` -> `["--filters", "filter_a", "--filters", "filter_b"]`)
377
378	# Parameters:
379	 - `pandoc_args : dict[str, Any]`
380
381	# Returns:
382	 - `list[str]`
383	"""
384	args: list[str] = []
385	for k, v in pandoc_args.items():
386		if isinstance(v, bool):
387			if v:
388				args.append(f"--{k}")
389		elif isinstance(v, str):
390			if k == "filter":
391				v = resolve_filter(v)
392			args.extend([f"--{k}", v])
393		elif isinstance(v, Iterable):
394			for x in v:
395				if k == "filter":
396					x = resolve_filter(str(x))
397				args.extend([f"--{k}", str(x)])
398		else:
399			raise ValueError(f"Invalid type for pandoc arg: {type(v) = } {v = }")
400
401	return args

404def dump_intermediate(
405	content: str,
406	intermediates_dir: Path | None,
407	fmt: str,
408	path: str,
409	subdir: str | None = None,
410) -> None:
411	"""Dump content to an intermediate file if intermediates_dir is specified"""
412	if intermediates_dir:
413		if subdir is None:
414			subdir = fmt
415		output_path: Path = intermediates_dir / fmt / f"{path}.{fmt}"

417		with open(output_path, "w", encoding="utf-8") as f:
418			f.write(content)

421def convert_single_markdown_file(
422	path: str,
423	output_root: Path,
424	doc: dict[str, Any],
425	docs: dict[str, dict[str, Any]],
426	jinja_env: Environment,
427	config: Config,
428	intermediates_dir: Path | None = None,
429) -> None:
430	"""Convert a single markdown document to HTML.
431
432	This function performs the full conversion pipeline for one document:
433	1. Extract frontmatter, body, and file metadata from the doc dict
434	2. Build context with frontmatter, config, all docs, and directory info
435	3. Render the markdown body with Jinja2
436	4. Convert rendered markdown to HTML with Pandoc
437	5. Apply the HTML template with the converted content
438	6. Optionally prettify the HTML output
439	7. Write the final HTML to the output directory
440
441	# Parameters:
442	 - `path : str` - relative path of the document (without .md extension)
443	 - `output_root : Path` - root directory for output (typically the config file's parent)
444	 - `doc : dict[str, Any]` - document dict with 'frontmatter', 'body', and 'file_meta' keys
445	 - `docs : dict[str, dict[str, Any]]` - dictionary of all documents in the site
446	 - `jinja_env : Environment` - Jinja2 environment for template rendering
447	 - `config : Config` - site configuration
448	 - `intermediates_dir : Path | None` - if provided, intermediate files are saved for debugging
449	"""
450	frontmatter: dict[str, Any] = doc.get("frontmatter", {})
451	if not isinstance(frontmatter, dict):
452		raise TypeError(f"Expected frontmatter to be dict, got {type(frontmatter)}")
453	body: str = doc.get("body", "")
454	if not isinstance(body, str):
455		raise TypeError(f"Expected body to be str, got {type(body)}")
456	file_meta: dict[str, Any] = doc.get("file_meta", {})
457	if not isinstance(file_meta, dict):
458		raise TypeError(f"Expected file_meta to be dict, got {type(file_meta)}")
459
460	# Get directory info for new template variables
461	file_dir: Path = Path(file_meta["path_raw"]).parent
462	file_dir_rel: str = str(Path(file_meta["path"]).parent)
463
464	context: dict[str, Any] = {
465		**frontmatter,
466		"frontmatter": frontmatter,
467		"file_meta": file_meta,
468		"config": config.serialize(),
469		"docs": docs,
470		# Docs matching by path prefix (original child_docs behavior)
471		"child_docs_dotlist": {
472			k: v for k, v in docs.items() if (k.startswith(path) and k != path)
473		},
474		# Docs in same folder (excluding current file)
475		"child_docs_folder": {
476			k: v
477			for k, v in docs.items()
478			if (str(Path(k).parent) == file_dir_rel and k != path)
479		},
480		# All files in the directory (filenames only)
481		"dir_files": [f.name for f in file_dir.iterdir() if f.is_file()],
482		# All subdirectories in the directory (names only)
483		"dir_subdirs": [d.name for d in file_dir.iterdir() if d.is_dir()],
484		# All files recursively (relative paths from dir)
485		"dir_contents_recursive": [
486			str(f.relative_to(file_dir)) for f in file_dir.rglob("*") if f.is_file()
487		],
488	}
489
490	dump_intermediate_partial: Callable[..., None] = functools.partial(
491		dump_intermediate,
492		intermediates_dir=intermediates_dir,
493		path=file_meta["path"],
494	)
495
496	# dump frontmatter to intermediates
497	dump_intermediate_partial(
498		content=str(frontmatter),
499		fmt="txt",
500		subdir="frontmatter_txt",
501	)
502	dump_intermediate_partial(
503		content=json.dumps(json_serialize(frontmatter)),
504		fmt="json",
505		subdir="frontmatter_json",
506	)
507
508	# Now, execute a template on the content with context
509	# Render Markdown content with Jinja2
510	rendered_md: str = render(
511		content=body,
512		context=context,
513		jinja_env=jinja_env,
514	)
515
516	dump_intermediate_partial(content=rendered_md, fmt="md")
517
518	# Convert Markdown to HTML using Pandoc
519	pandoc_args: list[str] = process_pandoc_args(
520		{
521			**config.__pandoc__,
522			**context["frontmatter"].get("__pandoc__", {}),
523		}
524	)
525
526	html_content: str = pypandoc.convert_text(
527		source=rendered_md,
528		to=config.pandoc_fmt_to,
529		format=config.pandoc_fmt_from,
530		extra_args=pandoc_args,
531	)
532
533	dump_intermediate_partial(content=html_content, fmt="html")
534
535	# Determine which HTML template to use
536	template_name: str = frontmatter.get(
537		"__template__",
538		config.default_template.as_posix(),
539	)
540
541	# Render final HTML
542	template: Template = jinja_env.get_template(template_name)
543	final_html: str = template.render({"__content__": html_content, **context})
544	if config.prettify:
545		final_html = str(
546			BeautifulSoup(final_html, "html.parser").prettify(formatter="minimal")
547		)
548
549	# Output HTML file
550	output_path: Path = output_root / config.output_dir / file_meta["path_html"]
551	output_path.parent.mkdir(parents=True, exist_ok=True)
552	with open(output_path, "w", encoding="utf-8") as f:
553		f.write(final_html)

556def convert_markdown_files(
557	docs: dict[str, dict[str, Any]],
558	jinja_env: Environment,
559	config: Config,
560	output_root: Path,
561	smart_rebuild: bool,
562	rebuild_time: float,
563	verbose: bool = True,
564	intermediates_dir: Path | None = None,
565) -> None:
566	"""Convert all markdown documents to HTML files.
567
568	Iterates through all documents and converts each to HTML. Supports smart
569	rebuild mode where only modified files are reprocessed.
570
571	# Parameters:
572	 - `docs : dict[str, dict[str, Any]]` - dictionary of documents from build_document_tree()
573	 - `jinja_env : Environment` - Jinja2 environment for template rendering
574	 - `config : Config` - site configuration
575	 - `output_root : Path` - root directory for output
576	 - `smart_rebuild : bool` - if True, skip files not modified since rebuild_time
577	 - `rebuild_time : float` - Unix timestamp of last build (for smart rebuild)
578	 - `verbose : bool` - if True, print progress information
579	 - `intermediates_dir : Path | None` - if provided, save intermediate files for debugging
580
581	# Raises:
582	 - `ConversionError` : if a single file fails to convert
583	 - `MultipleExceptions` : if multiple files fail to convert
584	"""
585	n_files: int = len(docs)
586	path: str
587	doc: dict[str, Any]
588	exceptions: dict[str, Exception] = {}
589	if verbose:
590		print(f"Converting {n_files} markdown files to HTML...")
591	for idx, (path, doc) in enumerate(docs.items()):
592		path_raw: str = doc["file_meta"]["path_raw"]
593		if smart_rebuild and Path(path_raw).stat().st_mtime <= rebuild_time:
594			if verbose:
595				print(f"\t({idx + 1:3} / {n_files})  [unmodified]  '{path_raw}'")
596		else:
597			if verbose:
598				print(f"\t({idx + 1:3} / {n_files})  [building..]  '{path_raw}'")
599
600			try:
601				convert_single_markdown_file(
602					path=path,
603					output_root=output_root,
604					doc=doc,
605					docs=docs,
606					jinja_env=jinja_env,
607					config=config,
608					intermediates_dir=intermediates_dir,
609				)
610			except Exception as e:
611				if isinstance(e, (KeyboardInterrupt, SystemExit)):
612					raise
613				exceptions[path_raw] = e
614				if verbose:
615					print(f"\t\t\033[91mERROR: could not convert '{path_raw}'\033[0m")
616	if exceptions:
617		first_key: str = next(iter(exceptions.keys()))
618		if len(exceptions) == 1:
619			raise ConversionError(
620				f"error converting file '{first_key}'",
621				n_failed=1,

53	"""Determine if a file should be copied based on include/exclude patterns.
54

56	- If file matches any include pattern → copy (wins over exclude)

79		return False

455	if not isinstance(body, str):
456		raise TypeError(f"Expected body to be str, got {type(body)}")
457	file_meta: dict[str, Any] = doc.get("file_meta", {})
458	if not isinstance(file_meta, dict):
459		raise TypeError(f"Expected file_meta to be dict, got {type(file_meta)}")

462	file_dir: Path = Path(file_meta["path_raw"]).parent
463	file_dir_rel: str = str(Path(file_meta["path"]).parent)
464

466		**frontmatter,
467		"frontmatter": frontmatter,
468		"file_meta": file_meta,
469		"config": config.serialize(),
470		"docs": docs,
471		# Docs matching by path prefix (original child_docs behavior)
472		"child_docs_dotlist": {
473			k: v for k, v in docs.items() if (k.startswith(path) and k != path)
474		},
475		# Docs in same folder (excluding current file)
476		"child_docs_folder": {
477			k: v
478			for k, v in docs.items()
479			if (str(Path(k).parent) == file_dir_rel and k != path)
480		},
481		# All files in the directory (filenames only)
482		"dir_files": [f.name for f in file_dir.iterdir() if f.is_file()],
483		# All subdirectories in the directory (names only)
484		"dir_subdirs": [d.name for d in file_dir.iterdir() if d.is_dir()],
485		# All files recursively (relative paths from dir)
486		"dir_contents_recursive": [
487			str(f.relative_to(file_dir)) for f in file_dir.rglob("*") if f.is_file()
488		],
489	}

492		dump_intermediate,

495	)

498	dump_intermediate_partial(

500		fmt="txt",
501		subdir="frontmatter_txt",

503	dump_intermediate_partial(

505		fmt="json",
506		subdir="frontmatter_json",
507	)
508
509	# Now, execute a template on the content with context
510	# Render Markdown content with Jinja2
511	rendered_md: str = render(
512		content=body,
513		context=context,
514		jinja_env=jinja_env,
515	)
516

518
519	# Convert Markdown to HTML using Pandoc
520	pandoc_args: list[str] = process_pandoc_args(
521		{
522			**config.__pandoc__,
523			**context["frontmatter"].get("__pandoc__", {}),
524		}
525	)
526
527	html_content: str = pypandoc.convert_text(
528		source=rendered_md,
529		to=config.pandoc_fmt_to,
530		format=config.pandoc_fmt_from,
531		extra_args=pandoc_args,
532	)
533
534	dump_intermediate_partial(content=html_content, fmt="html")
535
536	# Determine which HTML template to use
537	template_name: str = frontmatter.get(
538		"__template__",
539		config.default_template.as_posix(),
540	)
541
542	# Render final HTML
543	template: Template = jinja_env.get_template(template_name)
544	final_html: str = template.render({"__content__": html_content, **context})
545	if config.prettify:
546		final_html = str(
547			BeautifulSoup(final_html, "html.parser").prettify(formatter="minimal")
548		)
549
550	# Output HTML file
551	output_path: Path = output_root / config.output_dir / file_meta["path_html"]
552	output_path.parent.mkdir(parents=True, exist_ok=True)
553	with open(output_path, "w", encoding="utf-8") as f:
554		f.write(final_html)
