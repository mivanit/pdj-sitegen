 26def Plain_factory(val: str) -> dict:
 27	return {
 28		"t": "Plain",
 29		"c": [
 30			{"t": "Str", "c": val.strip()}

 36def table_cell_factory(val: str) -> list:
 37	return [

 43	]

 72def codeblock_process(key, value, format_, _):
 73	# figure out whether this block should be processed
 74	if not (key == "CodeBlock"):
 75		return None
 76
 77	[[ident, classes, lst_keyvals], code] = value
 78
 79	if "csv_table" not in classes:
 80		return None
 81
 82	# read the keyvals
 83	keyvals: dict = keyvals_process(lst_keyvals)
 84	header: bool = bool(int(keyvals.get("header", 1)))
 85	source: Optional[str] = keyvals.get("source")
 86	aligns: Optional[list[str]] = (
 87		list(keyvals.get("aligns", "")) if "aligns" in keyvals else None
 88	)
 89	caption: Optional[str] = keyvals.get("caption", None)
 90
 91	# read the csv source into a table
 92	if source is None:
 93		table_data = list(csv.reader(io.StringIO(code)))
 94	else:
 95		if os.path.isfile(source):
 96			table_data = list(csv.reader(open(source, "r")))
 97		else:
 98			raise Exception(f"csv source file not found: {source}")
 99
100	# validate the csv table
101	n_cols: int = len(table_data[0])
102	assert all(len(row) == n_cols for row in table_data), "csv table is not rectangular"
103
104	if aligns is None:
105		aligns = ["D" for _ in range(n_cols)]
106	else:
107		if len(aligns) == 1:
108			aligns = [aligns[0].upper() for _ in range(n_cols)]
109		elif len(aligns) == n_cols:
110			aligns = [aln.upper() for aln in aligns]
111		else:
112			raise Exception(f"aligns length mismatch: {aligns}")
113
114	if header:
115		row_header: list = table_data[0]
116		table_rows: list = table_data[1:]
117	else:
118		raise Exception("lack of header not supported")
119		row_header = []
120		table_rows = table_data
121
122	# write the table
123	return {
124		"t": "Table",
125		"c": [
126			# idk
127			emptyblock(),
128			# caption
129			[
130				None,
131				[] if caption is None else [Plain_factory(caption)],
132			],
133			# aligns
134			[
135				[
136					{"t": ALIGN_MAP[aln]},
137					{"t": "ColWidthDefault"},
138				]
139				for aln in aligns
140			],
141			# header
142			header_factory(row_header),
143			# rows
144			body_factory(table_rows),
145			# ???
146			[emptyblock(), []],
147		],
148	}
149

151def test_filter():
152	import json
153
154	with open(sys.argv[1]) as f:
155		data = json.load(f)
156	key = data["blocks"][0]["t"]
157	value = data["blocks"][0]["c"]
158	newdata = codeblock_process(key, value, "", "")
159	print(json.dumps(newdata, indent=2))
