 28def read_data_file(file_path: Path, fmt: Format | None = None) -> dict[str, Any]:
 29	"read a file from any of json, yaml, or toml"
 30	if fmt is None:
 31		suffix: str = file_path.suffix.lstrip(".")
 32		if suffix not in FORMAT_MAP:
 33			raise ValueError(
 34				f"Unknown file format: '.{suffix}'. Supported formats: {', '.join(FORMAT_MAP.keys())}"
 35			)
 36		fmt = FORMAT_MAP[suffix]
 37
 38	match fmt:
 39		case "yaml":
 40			with open(file_path, "r", encoding="utf-8") as f:
 41				return yaml.safe_load(f)
 42		case "json":
 43			with open(file_path, "r", encoding="utf-8") as f:
 44				return json.load(f)
 45		case "toml":
 46			with open(file_path, "rb") as f:
 47				return tomllib.load(f)
 48		case _:
 49			raise ValueError(f"Unsupported format: {fmt}")
 50

 52def emit_data_file(data: dict[str, Any], fmt: Format) -> str:
 53	"emit a file as json or yaml"
 54	match fmt:
 55		case "yaml":
 56			return yaml.safe_dump(data)
 57		case "json":
 58			return json.dumps(data, indent="\t")
 59		case "toml":
 60			raise NotImplementedError("Saving to TOML is not implemented.")
 61		case _:
 62			raise ValueError(f"Unsupported format: {fmt}")

 67) -> None:
 68	"save a file as json or yaml"

 77	emitted_data: str = emit_data_file(data, fmt)
 78	with open(file_path, "w", encoding="utf-8") as f:
 79		f.write(emitted_data)

 92@dataclass
 93class Config:
 94	"configuration for the site generator"
 95
 96	# paths
 97	content_dir: Path = field(default_factory=lambda: Path("content"))
 98	templates_dir: Path = field(default_factory=lambda: Path("templates"))
 99	default_template: Path = field(default_factory=lambda: Path("default.html.jinja2"))
100	intermediates_dir: Path | None = None
101	output_dir: Path = field(default_factory=lambda: Path("output"))
102	build_time_fname: Path = field(default_factory=lambda: Path(".build_time"))
103
104	# jinja2 settings and extra globals
105	jinja_env_kwargs: dict[str, Any] = field(default_factory=dict)
106	globals_: dict[str, Any] = field(default_factory=dict)
107
108	# whether to prettify html with bs4
109	prettify: bool = False
110
111	# content mirroring settings
112	# copy_include: patterns to include (empty = everything)
113	# copy_exclude: patterns to exclude (*.md excluded by default)
114	# if a file matches both, include wins
115	copy_include: list[str] = field(default_factory=list)
116	copy_exclude: list[str] = field(default_factory=lambda: ["*.md"])
117
118	# pandoc settings
119	__pandoc__: dict[str, Any] = field(default_factory=lambda: {"mathjax": True})
120	pandoc_fmt_from: str = "markdown+smart"
121	pandoc_fmt_to: str = "html"
122
123	@classmethod
124	def load(cls, data: dict[str, Any]) -> "Config":
125		"""Load Config from a dictionary."""
126		# Filter out __format__ key if present (legacy support)
127		filtered = {k: v for k, v in data.items() if k != "__format__"}
128
129		# Convert path strings to Path objects
130		for field_name in _PATH_FIELDS:
131			if field_name in filtered and filtered[field_name] is not None:
132				filtered[field_name] = Path(filtered[field_name])
133
134		return cls(**filtered)
135
136	def serialize(self) -> dict[str, Any]:
137		"""Serialize Config to a dictionary."""
138		result: dict[str, Any] = {"__format__": "Config(SerializableDataclass)"}
139
140		for field_name in self.__dataclass_fields__:
141			value = getattr(self, field_name)
142			if field_name in _PATH_FIELDS:
143				result[field_name] = value.as_posix() if value else None
144			else:
145				result[field_name] = value
146
147		return result
148
149	@classmethod
150	def read(cls, config_path: Path, fmt: Format | None = None) -> "Config":
151		return cls.load(read_data_file(config_path, fmt))

153	def as_str(self, fmt: Format) -> str:
154		return emit_data_file(self.serialize(), fmt)

156	def save(self, config_path: Path, fmt: Format | None = "json") -> None:
157		save_data_file(self.serialize(), config_path, fmt)
