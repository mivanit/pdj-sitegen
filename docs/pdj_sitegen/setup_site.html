pdj_sitegen.setup_site

CLI for scaffolding a new pdj-sitegen site.

Creates default configuration, templates, and content files to get started quickly. Run with python -m pdj_sitegen.setup_site [directory] to scaffold a new site.

 2
 3Creates default configuration, templates, and content files to get started quickly.
 4Run with `python -m pdj_sitegen.setup_site [directory]` to scaffold a new site.
 5"""

 7import argparse
 8import importlib.resources
 9from pathlib import Path
10
11import pdj_sitegen
12from pdj_sitegen.config import Config
13
14#: Default configuration instance used to determine output paths
15DEFAULT_CONFIG: Config = Config()
16
17#: Subdirectory for CSS and other static resources within content
18RESOURCES_DIR: Path = Path("resources")
19
20#: Mapping of source filenames (in package data) to their destination paths
21FILE_LOCATIONS: dict[str, Path] = {
22	"config.yml": Path("config.yml"),
23	"default.html.jinja2": DEFAULT_CONFIG.templates_dir / "default.html.jinja2",
24	"index.md": DEFAULT_CONFIG.content_dir / "index.md",
25	"style.css": DEFAULT_CONFIG.content_dir / RESOURCES_DIR / "style.css",
26	"syntax.css": DEFAULT_CONFIG.content_dir / RESOURCES_DIR / "syntax.css",
27}
28
29
30def setup_site(root: Path = Path(".")) -> None:
31	"""Scaffold a new pdj-sitegen site with default files.
32
33	Copies bundled template files from the package to the specified root directory,
34	creating the necessary directory structure.
35
36	# Parameters:
37	 - `root : Path`
38	   Root directory to create site files in. Defaults to current directory.
39
40	# Creates:
41	 - `config.yml` - default site configuration
42	 - `templates/default.html.jinja2` - default HTML template
43	 - `content/index.md` - sample index page
44	 - `content/resources/style.css` - basic stylesheet
45	 - `content/resources/syntax.css` - code syntax highlighting
46	"""
47	for file, path_rel in FILE_LOCATIONS.items():
48		path: Path = root / path_rel
49		if path.exists():
50			print(f"Skipping existing file: {path}")
51			continue
52
53		contents: str = (
54			importlib.resources.files(pdj_sitegen).joinpath("data", file).read_text()
55		)
56		path.parent.mkdir(parents=True, exist_ok=True)
57		with open(path, "w", encoding="utf-8") as f:
58			f.write(contents)
59
60
61def main() -> None:
62	"""CLI entry point for setup_site."""
63	arg_parser: argparse.ArgumentParser = argparse.ArgumentParser(
64		description="Set up a new pdj-sitegen site with default config and templates.",
65		epilog=(
66			"Creates:\n"
67			"  config.yml                         - default configuration\n"
68			"  templates/default.html.jinja2      - default HTML template\n"
69			"  content/index.md                   - sample index page\n"
70			"  content/resources/style.css        - basic stylesheet\n"
71			"  content/resources/syntax.css       - code syntax highlighting"
72		),
73		formatter_class=argparse.RawDescriptionHelpFormatter,
74	)
75	arg_parser.add_argument(
76		"root",
77		type=str,
78		nargs="?",
79		default=".",
80		help="root directory to create site files in (default: current directory)",
81	)
82	args: argparse.Namespace = arg_parser.parse_args()
83	setup_site(Path(args.root))
84
85
86if __name__ == "__main__":
87	main()
